107 /* Pass analyze_string a pointer to a char mxArray. Each element
108  in a char mxArray holds one 2-byte character (an mxChar);
109  analyze_string displays the contents of the input char mxArray
110  one row at a time. Since adjoining row elements are NOT stored in
111  successive indices, analyze_string has to do a bit of math to
112  figure out where the next letter in a string is stored. */
113 void analyze_string(const mxArray *string_array_ptr)
114 {
115  char *buf;
116  mwSize number_of_dimensions, buflen;
117  const mwSize *dims;
118  mwSize d, page, total_number_of_pages, elements_per_page;
119 
120  /* Allocate enough memory to hold the converted string. */
121  buflen = mxGetNumberOfElements(string_array_ptr) + 1;
122  buf = (char *)mxCalloc(buflen, sizeof(char));
123 
124  /* Copy the string data from string_array_ptr and place it into buf. */
125  if (mxGetString(string_array_ptr, buf, buflen) != 0)
126  mexErrMsgIdAndTxt( "MATLAB:explore:invalidStringArray",
127  "Could not convert string data.");
128 
129  /* Get the shape of the input mxArray. */
130  dims = mxGetDimensions(string_array_ptr);
131  number_of_dimensions = mxGetNumberOfDimensions(string_array_ptr);
132 
133  elements_per_page = dims[0] * dims[1];
134  /* total_number_of_pages = dims[2] x dims[3] x ... x dims[N-1] */
135  total_number_of_pages = 1;
136  for (d=2; d<number_of_dimensions; d++) {
137  total_number_of_pages *= dims[d];
138  }
139 
140  for (page=0; page < total_number_of_pages; page++) {
141  mwSize row;
142  /* On each page, walk through each row. */
143  for (row=0; row<dims[0]; row++) {
144  mwSize column;
145  mwSize index = (page * elements_per_page) + row;
146  mexPrintf("\t");
147  display_subscript(string_array_ptr, index);
148  mexPrintf(" ");
149 
150  /* Walk along each column in the current row. */
151  for (column=0; column<dims[1]; column++) {
152  mexPrintf("%c",buf[index]);
153  index += dims[0];
154  }
155  mexPrintf("\n");
156 
157  }
158 
159  }
160 }

163 /* Pass analyze_sparse a pointer to a sparse mxArray. A sparse mxArray
164  only stores its nonzero elements. The values of the nonzero elements
165  are stored in the pr and pi arrays. The tricky part of analyzing
166  sparse mxArray's is figuring out the indices where the nonzero
167  elements are stored. (See the mxSetIr and mxSetJc reference pages
168  for details. */
169 void analyze_sparse(const mxArray *array_ptr)
170 {
171  double *pr, *pi;
172  mwIndex *ir, *jc;
173  mwSize col, total=0;
174  mwIndex starting_row_index, stopping_row_index, current_row_index;
175  mwSize n;
176  char buf[100];
177  /* Get the starting positions of all four data arrays. */
178  pr = mxGetPr(array_ptr);
179  pi = mxGetPi(array_ptr);
180  ir = mxGetIr(array_ptr);
181  jc = mxGetJc(array_ptr);
182 
183  /* Display the nonzero elements of the sparse array. */
184  n = mxGetN(array_ptr);
185  for (col=0; col<n; col++) {
186  starting_row_index = jc[col];
187  stopping_row_index = jc[col+1];
188  if (starting_row_index == stopping_row_index) continue;
189  for (current_row_index = starting_row_index; current_row_index < stopping_row_index; current_row_index++){
190  if (mxIsComplex(array_ptr)) {
191  snprintf(buf, 100, "\t(%%%su,%%%su) = %%g+%%g i\n","z","z");
192  mexPrintf(buf,ir[current_row_index]+1,col+1, pr[total], pi[total]);
193  total++;
194  } else {
195  snprintf(buf, 100, "\t(%%%su,%%%su) = %%g\n","z","z");
196  mexPrintf(buf,ir[current_row_index]+1, col+1, pr[total++]);
197  }
198  }
199  }

451 /* Display the subscript associated with the given index. */
452 void display_subscript(const mxArray *array_ptr, mwSize index)
453 {
454  mwSize inner, subindex, total, d, q, number_of_dimensions;
455  mwSize *subscript;
456  const mwSize *dims;
457 
458  number_of_dimensions = mxGetNumberOfDimensions(array_ptr);
459  subscript = (mwSize*) mxCalloc(number_of_dimensions, sizeof(mwSize));
460  dims = mxGetDimensions(array_ptr);
461 
462  mexPrintf("(");
463  subindex = index;
464  for (d = number_of_dimensions-1; ; d--) { /* loop termination is at the end */
465 
466  for (total=1, inner=0; inner<d; inner++)
467  total *= dims[inner];
468 
469  subscript[d] = subindex / total;
470  subindex = subindex % total;
471  if (d == 0) {
472  break;
473  }
474  }
475 
476  for (q=0; q<number_of_dimensions-1; q++) {
477  mexPrintf("%d,", subscript[q] + 1);
478  }
479  mexPrintf("%d)", subscript[number_of_dimensions-1] + 1);
480 
481  mxFree(subscript);
482 }

510  strcat(shape_string, temp_string);
511  }

514  /* replace the last 'x' with a space */
515  shape_string[length_of_shape_string-1]='\0';

518  }
519  mexPrintf("Dimensions: %s\n", shape_string);
